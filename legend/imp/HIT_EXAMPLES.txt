  sleep 1
done < cards.txt

import requests
import json

base_url = "http://localhost:8000/hit.php"

cards = [
    "4111111111111111|12|2027|123",
    "5555555555554444|01|2026|456",
    "378282246310005|12|2025|1234"
]

address = "350, 5th Ave, New York, NY, 10118"
site = "https://shop.com"

for card in cards:
    params = {
        'cc': card,
        'address': address,
        'site': site,
        'format': 'json'
    }
    
    response = requests.get(base_url, params=params)
    result = response.json()
    
    if result['success']:
        for r in result['results']:
            print(f"Card: {r['card'][:6]}******{r['card'][-4:]}")
            print(f"Status: {r['status']}")
            print(f"Message: {r['message']}")
            print(f"Gateway: {r['gateway']}")
            print(f"Response Time: {r['response_time']}ms")
            print("-" * 50)
